Sloppy Mode [Non-Strict Mode]


<script>
function f1(){
    a=10;
    document.write("Sloppy Mode [Non-Strict Mode]"+"<br>")
    document.write("Value of a is : "+a+"<br>");
}
f1();
</script>

Strict Mode


<script>
"use strict";
function f2() {
    var b = 20;
    document.write("In Strict Mode" + "<br>");
    document.write("Value of b is : " + b + "<br>");
}
f2();
</script>
